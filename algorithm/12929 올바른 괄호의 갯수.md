## 아이디어
Backtracking(Brute Force)

모든 경우의 수($2^{28}$)를 탐색하되, 가지수가 너무 많아지므로 필요없는 가지를 친다(Backtracking, Pruning).

```java
// 답의 중복 제거를 위함
Set<String> answer = new HashSet<>();

/** 괄호를 전부 채웠다면 올바른 괄호인지 검증 */
boolean isValid(char[] parentheses) {
    int stack = 0;
    for (int i = 0; i < parentheses.length; i++) {
        char c = parentheses[i];
        if (c == '(') {
            stack++;
        } else if (c == ')') {
            if (stack <= 0) {
                return false;
            }
            stack--;
        }
    }
    return stack == 0;
}

/**
 * 재귀호출로 pruning 하며 완전 탐색
 * 
 * @param depth 현재 진행중 깊이
 * @param n 전체 괄호 개수
 * @param parentheses 현재 괄호 트리
 * @param left 사용한 '(' 개수
 * @param right 사용한 ')' 개수
 */
void recur(int depth, int n, char[] parentheses, int left, int right) {

    // 괄호를 전부 채웠다면
    if (depth == n) {
        String s = new String(parentheses);
        // 이전에 나오지 않은 괄호인지 확인하고 올바른 괄호인지 검증
        if (!answer.contains(s) && isValid(parentheses)) {
            answer.add(s);
        }
        return;
    }

    // 왼쪽 괄호의 여분이 남아있고 오른쪽 괄호보다 개수가 같거나 많아야 왼쪽 괄호를 현재 사용 가능
    if (left < n && left >= right) {
        parentheses[depth] = '(';
        recur(depth + 1, n, parentheses, left + 1, right);
    }
    // 오른쪽 괄호의 여분이 남아있고 왼쪽 괄호보다 개수가 적어야 오른쪽 괄호를 현재 사용 가능
    if (right < n && right < left) {
        parentheses[depth] = ')';
        recur(depth + 1, n, parentheses, left, right + 1);
    }
}

// 재귀함수 호출 & 결과 반환
public int solution(int n) {
    recur(0, n * 2, new char[n * 2], 0, 0);
    return answer.size();
}
```
