## 아이디어
Simulation(Implementation), Arithmetic(Math)

```python
def solution(a: int, b: int, n: int) -> int:
    answer = 0  # 빈병으로 받은 콜라
    while n >= a:   # 일단 콜라가 a개 이상은 있어야 받을 수 있음
        bottle = n // a * b     # 빈병으로 받을 수 있는 콜라
        answer += bottle        # 받을 수 있는 콜라 카운팅
        n %= a                  # 콜라를 받기 위해 필요한 만큼 마심
        n += bottle             # 한 턴이 끝난 후 남은 콜라의 개수
    return answer
```
